<script>
function noClick(){
  document.getElementById("ribboncat").style.display="none";
  document.getElementById("crycat").style.display="block";
}

function yesClick(){
  document.getElementById("ribboncat").style.display="none";
  document.getElementById("crycat").style.display="none";
  document.getElementById("envelope").style.display="block";
}

function openEnvelope(){
  document.getElementById("envelope").classList.add("open");
  document.body.classList.add("opened");

  setTimeout(()=>{
    document.getElementById("letter").style.display="block";
    startHearts();
    dropPetals();
  },800);
}

function startHearts(){
  setInterval(()=>{
    const h=document.createElement("div");
    h.className="heart";
    h.innerHTML="üíó";
    h.style.left=Math.random()*100+"vw";
    document.body.appendChild(h);
    setTimeout(()=>h.remove(),6000);
  },450);
}

function dropPetals(){
  for(let i=0;i<18;i++){
    setTimeout(()=>{
      const p=document.createElement("div");
      p.className="petal";
      p.innerHTML="üå∏";
      p.style.left=(100+Math.random()*100)+"px";
      document.getElementById("envelope").appendChild(p);
      setTimeout(()=>p.remove(),5000);
    },i*160);
  }
}
